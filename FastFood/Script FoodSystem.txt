CREATE TABLE[dbo].[TB_CLIENTE](
[I_COD_CLIENTE] INT IDENTITY(1, 1) NOT NULL,
[S_NM_CLIENTE] VARCHAR(35) NOT NULL,
[S_CPF_CLIENTE] VARCHAR(14) NOT NULL,
PRIMARY KEY CLUSTERED ([I_COD_CLIENTE] ASC)
);

CREATE TABLE[dbo].[TB_TIPO](
[I_COD_TIPO] INT IDENTITY(1, 1) NOT NULL,
[S_NM_TIPO] VARCHAR(15) NOT NULL,
PRIMARY KEY CLUSTERED ([I_COD_TIPO] ASC)
);

CREATE TABLE[dbo].[TB_FUNCIONARIO](
[I_COD_FUNCIONARIO] INT IDENTITY(1, 1) NOT NULL,
[S_UNM_FUNCIONARIO] VARCHAR(35) NOT NULL,
[S_PASS_FUNCIONARIO] VARCHAR(35) NOT NULL,
PRIMARY KEY CLUSTERED ([I_COD_FUNCIONARIO] ASC)
);

CREATE TABLE[dbo].[TB_PEDIDO](
[I_COD_PEDIDO] INT IDENTITY(1, 1) NOT NULL,
[I_COD_CLIENTE] INT NOT NULL,
[D_VLRTOT_PEDIDO] FLOAT NOT NULL,

FOREIGN KEY ([I_COD_CLIENTE]) REFERENCES [dbo].[TB_CLIENTE] ([I_COD_CLIENTE])
);

CREATE TABLE[dbo].[TB_ITEMPEDIDO](
[I_COD_ITEMPEDIDO] INT IDENTITY(1, 1) NOT NULL,
[I_COD_PEDIDO] INT NOT NULL,
[I_QTD_ITEMPEDIDO] INT NOT NULL,
[D_SUBVLR_ITEMPEDIDO] FLOAT NOT NULL,

FOREIGN KEY ([I_COD_PEDIDO]) REFERENCES [dbo].[TB_PEDIDO] ([I_COD_PEDIDO])
);

CREATE TABLE[dbo].[TB_PRODUTO](
[I_COD_PRODUTO] INT IDENTITY(1, 1) NOT NULL,
[I_COD_TIPO] INT NOT NULL,
[S_NM_PRODUTO] VARCHAR(35) NOT NULL,
[S_DESC_PRODUTO] VARCHAR(100) NOT NULL,
[D_VLRUNI_PRODUTO] FLOAT NOT NULL,

FOREIGN KEY ([I_COD_TIPO]) REFERENCES [dbo].[TB_TIPO] ([I_COD_TIPO])
);

CREATE TABLE[dbo].[TB_ITEMCOMBO](
[I_COD_ITEMCOMBO] INT IDENTITY(1, 1) NOT NULL,
[I_COD_COMBO] INT NOT NULL,
[I_COD_PRODUTO] INT NOT NULL,
[I_QTD_PRODUTO] INT NOT NULL,

);
